package deadlock.d1;
public class App{
    public static void main(String args[]){
        final DeadlockExample dex=new DeadlockExample();
        Thread t1=new Thread(new Runnable(){
            public void run(){
                try{
dex.firstThread();
                }
                catch(IntrruptedException e){
                    e.printStackTrace();
                }
            }
        });


        Thread t2=new Thread(new Runnable(){
            public void run(){
                try{
dex.secondThread();
                }
                catch(IntrruptedException e){
                    e.printStackTrace();
                }
            }
        });
        t1.start();
        t2.start();
        t1.join();
        t2.join();
        dex.finished();
    }
}
